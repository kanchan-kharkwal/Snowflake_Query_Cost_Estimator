CREATE OR REPLACE PROCEDURE AKASX.PUBLIC.run_benchmark_queries()
RETURNS STRING
LANGUAGE SQL
AS
$$
DECLARE
    i INTEGER DEFAULT 0;
    query_cursor CURSOR FOR 
        SELECT query_text 
        FROM AKASX.PUBLIC.query_texts
        WHERE active_flag =1
        ORDER BY num;
BEGIN
    -- Loop over all query texts
    FOR rec IN query_cursor DO
        EXECUTE IMMEDIATE rec.query_text;
        i := i + 1;
    END FOR;

    RETURN 'Executed ' || TO_VARCHAR(i) || ' queries.';
END;
$$;
